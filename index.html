📟 ПОЛНЫЙ КОД МЕТЕОСТАНЦИИ

#include <ESP8266WiFi.h>
#include <WiFiClientSecure.h>
#include <UniversalTelegramBot.h>
#include <DHT.h>

// ========== ТВОИ ДАННЫЕ ==========
const char* ssid = "ваш Wi-Fi";
const char* password = "ваш пароль";
#define BOT_TOKEN "токен_бота"

// ID разрешенных пользователей
const String allowedUsers[] = {
  "1234567890",
  "123456789",
  "1234567890"
};
const int totalUsers = 3;
// =================================

#define DHTPIN 2
#define DHTTYPE DHT22

DHT dht(DHTPIN, DHTTYPE);
WiFiClientSecure client;
UniversalTelegramBot bot(BOT_TOKEN, client);

float temperature = 0;
float humidity = 0;
unsigned long lastTime = 0;

bool isAllowed(String chat_id) {
  for(int i = 0; i < totalUsers; i++) {
    if(allowedUsers[i] == chat_id) return true;
  }
  return false;
}

void setup() {
  Serial.begin(115200);
  dht.begin();
  client.setInsecure();
  
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  
  bot.sendMessage("404415869", "꧁༺ *БОТ ЗАПУЩЕН* ༻꧂", "Markdown");
}

void loop() {
  if(millis() - lastTime > 5000) {
    float t = dht.readTemperature();
    float h = dht.readHumidity();
    if(!isnan(t) && !isnan(h)) {
      temperature = t;
      humidity = h;
    }
    
    int msgCount = bot.getUpdates(bot.last_message_received + 1);
    
    for(int i = 0; i < msgCount; i++) {
      String chat = bot.messages[i].chat_id;
      String text = bot.messages[i].text;
      String name = bot.messages[i].from_name;
      
      if(!isAllowed(chat)) {
        bot.sendMessage(chat, "⛔️ *НЕТ ДОСТУПА*", "Markdown");
        continue;
      }
      
      if(text == "/start") {
        String msg = "꧁༺ *ДОБРО ПОЖАЛОВАТЬ* ༻꧂\n\n";
        msg += "*ПРИВЕТ, " + name + "*\n";
        msg += "🌡 /temp - температура\n";
        msg += "💧 /hum - влажность\n";
        msg += "📊 /all - все данные";
        bot.sendMessage(chat, msg, "Markdown");
      }
      
      else if(text == "/temp") {
        String msg = "🌡 *Температура:* " + String(temperature, 1) + "°C";
        bot.sendMessage(chat, msg, "Markdown");
      }
      
      else if(text == "/hum") {
        String msg = "💧 *Влажность:* " + String(humidity, 1) + "%";
        bot.sendMessage(chat, msg, "Markdown");
      }
      
      else if(text == "/all") {
        String msg = "📊 *Данные метеостанции*\n\n";
        msg += "🌡 Температура: " + String(temperature, 1) + "°C\n";
        msg += "💧 Влажность: " + String(humidity, 1) + "%";
        bot.sendMessage(chat, msg, "Markdown");
      }
    }
    
    lastTime = millis();
  }
}

#include <ESP8266WiFi.h>

const char* ssid = "ваш Wi-Fi";
const char* password = "ваш пароль";

const int redPin = D1;
const int greenPin = D2;
const int bluePin = D3;

WiFiServer server(80);

bool redState = false, greenState = false, blueState = false;

void setup() {
  Serial.begin(115200);
  
  pinMode(redPin, OUTPUT);
  pinMode(greenPin, OUTPUT);
  pinMode(bluePin, OUTPUT);
  
  digitalWrite(redPin, LOW);
  digitalWrite(greenPin, LOW);
  digitalWrite(bluePin, LOW);
  
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  
  Serial.println("\\nWiFi подключен!");
  Serial.print("IP адрес: ");
  Serial.println(WiFi.localIP());
  
  server.begin();
}

void loop() {
  WiFiClient client = server.available();
  if (!client) return;
  
  while(!client.available()) delay(1);
  
  String request = client.readStringUntil('\\r');
  client.flush();
  
  if (request.indexOf("/red=on") != -1) { digitalWrite(redPin, HIGH); redState = true; }
  if (request.indexOf("/red=off") != -1) { digitalWrite(redPin, LOW); redState = false; }
  if (request.indexOf("/green=on") != -1) { digitalWrite(greenPin, HIGH); greenState = true; }
  if (request.indexOf("/green=off") != -1) { digitalWrite(greenPin, LOW); greenState = false; }
  if (request.indexOf("/blue=on") != -1) { digitalWrite(bluePin, HIGH); blueState = true; }

#include <ESP8266WiFi.h>

const char* ssid = "ваш Wi-Fi";
const char* password = "ваш пароль";

const int redPin = D1;
const int greenPin = D2;
const int bluePin = D3;

WiFiServer server(80);

int redValue = 0, greenValue = 0, blueValue = 0;

void setup() {
  Serial.begin(115200);
  
  pinMode(redPin, OUTPUT);
  pinMode(greenPin, OUTPUT);
  pinMode(bluePin, OUTPUT);
  
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) delay(500);
  
  server.begin();
}

void loop() {
  WiFiClient client = server.available();
  if (!client) return;
  
  while(!client.available()) delay(1);
  
  String request = client.readStringUntil('\\r');
  client.flush();
  
  if (request.indexOf("/red=") != -1) {
    int pos = request.indexOf("/red=") + 5;
    String val = request.substring(pos, request.indexOf(" ", pos));
    redValue = val.toInt();
    analogWrite(redPin, redValue);
  }
  
  if (request.indexOf("/green=") != -1) {
    int pos = request.indexOf("/green=") + 7;
    String val = request.substring(pos, request.indexOf(" ", pos));
    greenValue = val.toInt();
    analogWrite(greenPin, greenValue);
  }
  
  if (request.indexOf("/blue=") != -1) {
    int pos = request.indexOf("/blue=") + 6;
    String val = request.substring(pos, request.indexOf(" ", pos));
    blueValue = val.toInt();
    analogWrite(bluePin, blueValue);
  }
  
  client.println("HTTP/1.1 200 OK");
  client.println("Content-Type: text/html");

#include <ESP8266WiFi.h>

const char* ssid = "ваш Wi-Fi";
const char* password = "ваш пароль";

const int redPin = D1;
const int greenPin = D2;
const int bluePin = D3;

WiFiServer server(80);

void setColor(int r, int g, int b) {
  analogWrite(redPin, r);
  analogWrite(greenPin, g);
  analogWrite(bluePin, b);
}

void setup() {
  Serial.begin(115200);
  
  pinMode(redPin, OUTPUT);
  pinMode(greenPin, OUTPUT);
  pinMode(bluePin, OUTPUT);
  
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) delay(500);
  
  server.begin();
}

void loop() {
  WiFiClient client = server.available();
  if (!client) return;
  
  while(!client.available()) delay(1);
  
  String request = client.readStringUntil('\\r');
  client.flush();
  
  if (request.indexOf("/color=red") != -1) setColor(1023, 0, 0);
  if (request.indexOf("/color=green") != -1) setColor(0, 1023, 0);
  if (request.indexOf("/color=blue") != -1) setColor(0, 0, 1023);
  if (request.indexOf("/color=yellow") != -1) setColor(1023, 1023, 0);
  if (request.indexOf("/color=cyan") != -1) setColor(0, 1023, 1023);
  if (request.indexOf("/color=magenta") != -1) setColor(1023, 0, 1023);
  if (request.indexOf("/color=white") != -1) setColor(1023, 1023, 1023);
  if (request.indexOf("/color=off") != -1) setColor(0, 0, 0);
  
  client.println("HTTP/1.1 200 OK");
  client.println("Content-Type: text/html");
