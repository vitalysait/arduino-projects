📟 КОД МЕТЕОСТАНЦИИ С TELEGRAM

#include <ESP8266WiFi.h>
#include <WiFiClientSecure.h>
#include <UniversalTelegramBot.h>
#include <DHT.h>

const char* ssid = "ваш Wi-Fi";
const char* password = "ваш пароль";
#define BOT_TOKEN "токен_бота"
#define DHTPIN 4  // D2
#define DHTTYPE DHT22

DHT dht(DHTPIN, DHTTYPE);
WiFiClientSecure client;
UniversalTelegramBot bot(BOT_TOKEN, client);

float temperature = 0;
float humidity = 0;
unsigned long lastTime = 0;

void setup() {
  Serial.begin(115200);
  dht.begin();
  client.setInsecure();
  
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) delay(500);
}

void loop() {
  if(millis() - lastTime > 5000) {
    float t = dht.readTemperature();
    float h = dht.readHumidity();
    if(!isnan(t) && !isnan(h)) {
      temperature = t;
      humidity = h;
    }
    
    int msgCount = bot.getUpdates(bot.last_message_received + 1);
    
    for(int i = 0; i < msgCount; i++) {
      String chat = bot.messages[i].chat_id;
      String text = bot.messages[i].text;
      
      if(text == "/temp") {
        bot.sendMessage(chat, "🌡 Температура: " + String(temperature, 1) + "°C", "");
      }
      else if(text == "/hum") {
        bot.sendMessage(chat, "💧 Влажность: " + String(humidity, 1) + "%", "");
      }
    }
    lastTime = millis();
  }
}

#include <ESP8266WiFi.h>

const char* ssid = "ваш Wi-Fi";
const char* password = "ваш пароль";

const int redPin = D1;
const int greenPin = D2;
const int bluePin = D3;

WiFiServer server(80);

void setup() {
  Serial.begin(115200);
  
  pinMode(redPin, OUTPUT);
  pinMode(greenPin, OUTPUT);
  pinMode(bluePin, OUTPUT);
  
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) delay(500);
  
  server.begin();
}

void loop() {
  WiFiClient client = server.available();
  if (!client) return;
  
  String request = client.readStringUntil('\r');
  client.flush();
  
  if (request.indexOf("/red=on") != -1) digitalWrite(redPin, HIGH);
  if (request.indexOf("/red=off") != -1) digitalWrite(redPin, LOW);
  
  client.println("HTTP/1.1 200 OK");
  client.println("Content-Type: text/html");
  client.println("");
  client.println("");
  client.println("
RGB
");
  client.println("🔴 ON
");
  client.println("⚫ OFF");
  client.println("");
}

📟 2. ESP32 - Сканер Wi-Fi сетей

#include <WiFi.h>

void setup() {
  Serial.begin(115200);
  WiFi.mode(WIFI_STA);
  WiFi.disconnect();
  delay(100);
}

void loop() {
  Serial.println("Сканирование...");
  int n = WiFi.scanNetworks();
  for (int i = 0; i < n; i++) {
    Serial.print(i + 1);
    Serial.print(": ");
    Serial.print(WiFi.SSID(i));
    Serial.print(" (");
    Serial.print(WiFi.RSSI(i));
    Serial.println(" dBm)");
  }
  delay(5000);
}

#include <WiFi.h>

const char* ssid = "ваш Wi-Fi";
const char* password = "ваш пароль";

WiFiServer server(80);

void setup() {
  Serial.begin(115200);
  pinMode(2, OUTPUT);
  
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) delay(500);
  
  server.begin();
}

void loop() {
  WiFiClient client = server.available();
  if (!client) return;
  
  String request = client.readStringUntil('\r');
  client.flush();
  
  if (request.indexOf("/on") != -1) digitalWrite(2, HIGH);
  if (request.indexOf("/off") != -1) digitalWrite(2, LOW);
  
  client.println("HTTP/1.1 200 OK");
  client.println("Content-Type: text/html");
  client.println("");
  client.println("<!DOCTYPE HTML><html>");
  client.println("<body><h1>ESP32</h1>");
  client.println("<a href='/on'>ВКЛ</a><br>");
  client.println("<a href='/off'>ВЫКЛ</a>");
  client.println("</body></html>");
}
